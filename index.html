Single select



Model: 

<multiselect
    ng-model="multiselectModel[0]"
    options="c for c in multiselectOptions"
    data-multiple="false"
></multiselect>

Multi select



Model: 

<multiselect
    ng-model="multiselectModel[0]"
    options="c for c in multiselectOptions"
    data-multiple="true"
></multiselect>

app.js:

angular.module("angular-bootstrap-multiselect-test", ["ui.multiselect"]).controller("index", function($scope) {
    $scope.multiselectModel = ["", []];
    $scope.multiselectOptions = [
        "Apples",
        "Bananas",
        "Peaches"
    ];
});
